
CREATE TABLE Board
(
	BoardID               NUMBER  NOT NULL ,
	BoardName             VARCHAR2(20)  NULL ,
	BoardSubject          VARCHAR2(20)  NULL ,
	BoardContent          VARCHAR2(4000)  NULL ,
	FileName              VARCHAR2(20)  NULL ,
	ReplyRef              NUMBER  NULL ,
	ReplyDepth            NUMBER  NULL ,
	ReplySeq              NUMBER  NULL ,
	CreateDate            DATE  NULL ,
	UpdateDate            DATE  NULL ,
	ReadCount             NUMBER  NULL ,
	Notice                VARCHAR2(20)  NULL ,
	UserID                VARCHAR2(20)  NOT NULL 
);



CREATE UNIQUE INDEX XPKBoard ON Board
(BoardID  ASC);
